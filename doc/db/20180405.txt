CREATE TABLE `base_introduce` (
  `id` varchar(64) NOT NULL COMMENT '编号',
  `create_by` varchar(64) NOT NULL COMMENT '创建者',
  `create_date` datetime NOT NULL COMMENT '创建时间',
  `update_by` varchar(64) NOT NULL COMMENT '更新者',
  `update_date` datetime NOT NULL COMMENT '更新时间',
  `remarks` varchar(255) DEFAULT NULL COMMENT '备注信息',
  `del_flag` char(1) NOT NULL DEFAULT '0' COMMENT '删除标记',
  `tital` varchar(255) DEFAULT NULL COMMENT '标题',
  `index` varchar(24) DEFAULT NULL COMMENT '顺序',
  `path` varchar(255) DEFAULT NULL COMMENT '图片',
  `user` varchar(64) DEFAULT NULL COMMENT '用户主键',
  PRIMARY KEY (`id`),
  KEY `base_introduce_del_flag` (`del_flag`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='介绍';



CREATE TABLE `base_news` (
  `id` varchar(64) NOT NULL COMMENT '编号',
  `create_by` varchar(64) NOT NULL COMMENT '创建者',
  `create_date` datetime NOT NULL COMMENT '创建时间',
  `update_by` varchar(64) NOT NULL COMMENT '更新者',
  `update_date` datetime NOT NULL COMMENT '更新时间',
  `remarks` varchar(255) DEFAULT NULL COMMENT '备注信息',
  `del_flag` char(1) NOT NULL DEFAULT '0' COMMENT '删除标记',
  `title` varchar(255) DEFAULT NULL COMMENT '标题',
  `path` varchar(255) DEFAULT NULL COMMENT '图片路径',
  `content` varchar(3000) DEFAULT NULL,
  `user` varchar(64) DEFAULT NULL COMMENT '用户主键',
  PRIMARY KEY (`id`),
  KEY `base_new_del_flag` (`del_flag`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='我的资讯';
